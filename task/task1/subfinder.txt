Run (PowerShell; from repo root)
New-Item -ItemType Directory -Force temp/agent1 | Out-Null
subfinder -d <domain> -silent -all -recursive -rl 10 -max-time 9 -timeout 30 -o temp/agent1/subfinder.txt 2>&1 | Tee-Object temp/agent1/subfinder.log

Merge into passive list (dedupe)
Get-Content temp/agent1/subfinder.txt |
  ForEach-Object { $_.Trim().ToLower().TrimEnd('.') } |
  Where-Object { $_ } |
  Sort-Object -Unique |
  Set-Content -Encoding utf8 temp/agent1/list_1_passive.txt
